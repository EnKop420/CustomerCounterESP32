228bool removeLatestEntryOnDate(const char * path, String targetDate) {
229 // Open the CSV file in read mode
230 File file = LittleFS.open(path, FILE_READ);
231 if(!file){
232 Serial.println("- failed to open file for writing");
233 return false;
234 }
235
236 // Create a temporary file to store the modified data
237 File tempFile = LittleFS.open("/temp.csv", FILE_WRITE);
238 if (!tempFile) {
239 Serial.println("Failed to open temporary file for writing");
240 file.close();
241 return false;
242 }
243
244 // Read the CSV file line by line
245 String line;
246 bool foundLatest = false;
247 while (file.available()) {
248 line = file.readStringUntil('\n');
249
250 // Extract date from the line
251 int commaIndex1 = line.indexOf(',');
252 int commaIndex2 = line.indexOf(',', commaIndex1 + 1);
253 String currentDate = line.substring(commaIndex1 + 1, commaIndex2);
254
255 // If the current date matches the target date and the latest flag is not set,
256 // skip writing this line to the temporary file
257 if (currentDate == targetDate && !foundLatest) {
258 foundLatest = true;
259 continue;
260 }
261
262 // Write the line to the temporary file
263 tempFile.println(line);
264 }
265
266 // Close both files
267 file.close();
268 tempFile.close();
269
270 // Delete the original CSV file
271 LittleFS.remove(path);
272
273 // Rename the temporary file to the original CSV file
274 LittleFS.rename("/temp.csv", path);
275
276 Serial.println("Latest entry on " + targetDate + " removed successfully!");
277 return true;

286bool removeLinesWithDate(const char* path, const String& inputDate) {
287 // Open the CSV file for reading
288 File file = LittleFS.open(path, "r");
289 if (!file) {
290 Serial.println("Failed to open file for reading");
291 return false;
292 }
293
294 // Temporary storage for modified content
295 String modifiedContent = "";
296
297 // Read file line-by-line
298 while (file.available()) {
299 String line = file.readStringUntil('\n');
300
301 // Skip empty lines or header (optional based on CSV structure)
302 if (line.length() == 0 || line.startsWith("customer")) {
303 modifiedContent += line + "\n";
304 continue;
305 }
306
307 // Find the first comma to get to the date part
308 int firstCommaIdx = line.indexOf(',');
309 if (firstCommaIdx == -1) {
310 Serial.println("Malformed CSV line; skipping: " + line);
311 continue; // Skip lines without a comma
312 }
313
314 // Extract the date (between the first and second comma)
315 int secondCommaIdx = line.indexOf(',', firstCommaIdx + 1);
316 if (secondCommaIdx == -1) {
317 Serial.println("Malformed CSV line; skipping: " + line);
318 continue; // Skip lines without a second comma
319 }
320
321 // The date part is between the first and second comma
322 String date = line.substring(firstCommaIdx + 1, secondCommaIdx);
323 date.trim(); // Remove any extra whitespace around the date
324
325 // Only add lines to modified content if the date doesn't match inputDate
326 if (date != inputDate) {
327 modifiedContent += line + "\n";
328 }
329 }
330
331 file.close(); // Close the file after reading
332
333 // Re-open the file for writing (overwrites existing content)
334 file = LittleFS.open(path, "w");
335 if (!file) {
336 Serial.println("Failed to open file for writing");
337 return false;
338 }
339
340 // Write the modified content back to the file
341 file.print(modifiedContent);
342 file.close();
343
344 Serial.println("Lines with date " + inputDate + " have been removed from the CSV.");
345 return true;
346}

412String countDates(const char* csv) {
413 const int MAX_DATES = 50; // Maximum expected unique dates
414 String dates[MAX_DATES]; // Array to store dates
415 int counts[MAX_DATES] = {0}; // Array to count occurrences
416 int uniqueDateCount = 0; // Counter for unique dates
417
418 String csvString = String(csv);
419 int startIdx = 0;
420 while (true) {
421 int lineEndIdx = csvString.indexOf('\n', startIdx); // Find end of line
422 if (lineEndIdx == -1) break; // Break if no more lines
423
424 String line = csvString.substring(startIdx, lineEndIdx); // Extract line
425 startIdx = lineEndIdx + 1;
426
427 if (line.length() == 0 || line.startsWith("customer")) continue; // Skip empty or header lines
428
429 // Extract date from the line
430 int commaIdx = line.indexOf(',');
431 int secondCommaIdx = line.indexOf(",", commaIdx + 1);
432 String date = line.substring(commaIdx + 1, secondCommaIdx);
433
434
435 // Check if the date is already in the list
436 bool found = false;
437 for (int i = 0; i < uniqueDateCount; i++) {
438 if (dates[i] == date) {
439 counts[i]++; // Increment count if date is found
440 found = true;
441 break;
442 }
443 }
444
445 // Add new date if not found
446 if (!found) {
447 dates[uniqueDateCount] = date;
448 counts[uniqueDateCount] = 1;
449 uniqueDateCount++;
450 }
451 }
452
453 // Convert to JSON
454 JsonDocument jsonDoc;
455 for (int i = 0; i < uniqueDateCount; i++) {
456 jsonDoc[dates[i]] = counts[i];
457 }
458
459 String jsonOutput;
460 serializeJson(jsonDoc, jsonOutput);
461 return jsonOutput; // Store each date with its count
462}
